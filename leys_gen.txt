# Generate Server's keys
openssl genrsa -out server_priv.pem 2048
openssl rsa -in server_priv.pem -pubout -out server_pub.pem

# Generate Client's keys
openssl genrsa -out client_priv.pem 2048
openssl rsa -in client_priv.pem -pubout -out client_pub.pem



# Generate private key
openssl genpkey -algorithm ED25519 -out server_sk.pem

# Extract public key
openssl pkey -in server_sk.pem -pubout -out server_pk.pem


# Generate private key
openssl genpkey -algorithm ED25519 -out client_sk.pem

# Extract public key
openssl pkey -in client_sk.pem -pubout -out client_pk.pem



# Server private key raw (64 bytes)
openssl pkey -in server_sk.pem -outform DER | tail -c 64 > server_ed25519_sk.bin

# Server public key raw (32 bytes)
openssl pkey -pubin server_pk.pem -outform DER | tail -c 32 > server_ed25519_pk.bin

# Client private key raw (64 bytes)
openssl pkey -in client_sk.pem -outform DER | tail -c 64 > client_ed25519_sk.bin

# Client public key raw (32 bytes)
openssl pkey -pubin client_pk.pem -outform DER | tail -c 32 > client_ed25519_pk.bin
